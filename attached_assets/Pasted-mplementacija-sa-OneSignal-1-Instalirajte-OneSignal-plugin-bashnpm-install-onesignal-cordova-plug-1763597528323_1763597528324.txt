4. Inicijalizacija u va≈°oj aplikaciji:
typescriptimport { Capacitor } from '@capacitor/core';

// OneSignal setup
const initializeOneSignal = () => {
  if (!Capacitor.isNativePlatform()) {
    console.log('‚ö†Ô∏è Not running on native platform');
    return;
  }

  // @ts-ignore - OneSignal je global objekat
  const OneSignal = window.plugins?.OneSignal;
  
  if (!OneSignal) {
    console.error('‚ùå OneSignal plugin not found');
    return;
  }

  console.log('üîî Initializing OneSignal...');

  // Inicijalizuj OneSignal
  OneSignal.setAppId('VA≈†_ONESIGNAL_APP_ID'); // Dobijete sa OneSignal dashboard-a

  // Omoguƒái notifikacije
  OneSignal.promptForPushNotificationsWithUserResponse((accepted) => {
    console.log('Push notifications accepted:', accepted);
  });

  // Slu≈°aj kada stigne notifikacija
  OneSignal.setNotificationOpenedHandler((jsonData) => {
    console.log('üîî Notification opened:', jsonData);
    
    // Ovde rukujte akcijom (npr. otvori odreƒëenu reklamaciju)
    const data = jsonData.notification.additionalData;
    if (data?.reklamacija_id) {
      // Navigirajte do reklamacije
      window.location.href = `/reklamacija/${data.reklamacija_id}`;
    }
  });

  // Dobij Player ID (koristite ga umesto FCM tokena)
  OneSignal.getDeviceState((state) => {
    console.log('üì± OneSignal Player ID:', state.userId);
    // Saƒçuvajte userId u bazi umesto FCM tokena
    savePlayerIdToDatabase(state.userId);
  });
};

// Pokrenite pri startu
document.addEventListener('deviceready', initializeOneSignal);

// Fallback
if (document.readyState === 'complete') {
  setTimeout(initializeOneSignal, 1000);
}

6. Backend integracija:
Umesto FCM API-ja, koristite OneSignal API:
javascript// Node.js primer
const axios = require('axios');

async function sendNotification(playerIds, title, message, data = {}) {
  const notification = {
    app_id: 'VA≈†_ONESIGNAL_APP_ID',
    include_player_ids: playerIds, // Array player ID-eva
    headings: { en: title },
    contents: { en: message },
    data: data, // Extra data (reklamacija_id, itd.)
    android_sound: 'notification_sound', // Bez .wav ekstenzije
    android_channel_id: 'reklamacije-alert',
    priority: 10
  };

  try {
    const response = await axios.post(
      'https://onesignal.com/api/v1/notifications',
      notification,
      {
        headers: {
          'Content-Type': 'application/json',
          'Authorization': 'Basic VA≈†_REST_API_KEY' // Sa OneSignal dashboard-a
        }
      }
    );
    console.log('‚úÖ Notification sent:', response.data);
  } catch (error) {
    console.error('‚ùå Error sending notification:', error);
  }
}

// Primer upotrebe:
